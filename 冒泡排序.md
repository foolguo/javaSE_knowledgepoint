public static void bubbleSort(int[] array){
        int n=array.length;
        if(n<=1){//如果数组不存在或者只有一个元素，直接返回
            return ;
        }else{
            //控制排序次数
            for (int i=0;i<n;i++){
                //设置标志
                int flag=0;
                //进行内部的两两交换
                for(int j=0;j<n-i-1;j++){
                    if(array[j]>array[j+1]){
                        int temp=array[j];
                        array[j]=array[j+1];
                        array[j+1]=temp;
                        flag=1;
                    }
                }
                if(flag==0){//如果此时标志还为0，说明序列本身有序，不需要进行排序
                    break;
                }
            }
        }
